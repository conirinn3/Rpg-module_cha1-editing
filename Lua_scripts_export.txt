local session_notifications_manager = {}

function addNotification(triggerId, message)
    if not session_notifications_manager[triggerId] then
        session_notifications_manager[triggerId] = {}
    end
    table.insert(session_notifications_manager[triggerId], message)
end

function recordTemporaryChange(triggerId, key, value)
    local sourceIndex = getChatLength(triggerId) - 1

    local delta = getState(triggerId, "temporary_state_delta") or {}

    -- 인덱스가 변경되었을 때만 델타를 초기화
    if delta.sourceIndex ~= sourceIndex then
        delta = {
            data = {},
            sourceIndex = sourceIndex
        }
    end

    delta.data[key] = value
    setState(triggerId, "temporary_state_delta", delta)
    print(string.format("Recorded temporary change: %s = %s at source index %d", key, tostring(value), sourceIndex))
end

-- '%' 문자를 'percent'라는 단어로 안전하게 교체
local function PerscentReplace(originalString, placeholder, replacementValue)
    if type(replacementValue) ~= "string" then
        return originalString
    end
    local sanitizedValue = replacementValue:gsub("%%", "percent")
    return originalString:gsub(placeholder, sanitizedValue)
end

    local stateKey = ""
    local maxVal = nil
    
    if target == "LuckyCoin" then stateKey = "Lucky_Coin_Count"
    elseif target == "Adrenaline" then stateKey = "Adrenaline_Token_Count"; maxVal = 3
    elseif target == "Preparation" then stateKey = "Preparation_Token_Count"; maxVal = 5
    elseif target == "AllOrNothing" then stateKey = "All_or_Nothing_Token_Count"; maxVal = 3
    elseif target == "NoTime" then stateKey = "No_Time_to_Spare_Token_Count"; maxVal = 2
    elseif target == "Tempo_Melee" then stateKey = "State_Tempo_Melee_Lost"
    elseif target == "Tempo_Archery" then stateKey = "State_Tempo_Archery_Lost"
    elseif target == "Fearsome" then stateKey = "State_Fearsome_Presence_Active"
    elseif target == "Reversal" then stateKey = "Reversal_Token_Count"
    elseif target == "FailureCount" then stateKey = "Failure_Accumulation_Count"
    else return 0 end

    local currentVal = getState(triggerId, stateKey) or 0
    local newVal = currentVal

    local amount = value or 1
    if action == "set" then newVal = amount
    elseif action == "add" then newVal = currentVal + amount
    elseif action == "sub" or action == "use" then newVal = currentVal - amount end

    if target == "LuckyCoin" then newVal = math.max(0, newVal)
    elseif (target == "Adrenaline" or target == "Preparation" or target == "AllOrNothing" or target == "NoTime") and maxVal then 
        newVal = math.max(0, math.min(newVal, maxVal)) 
    end
    
    -- Reversal 토큰과 실패 카운트는 음수가 되지 않도록 방지
    if target == "Reversal" or target == "FailureCount" then
        newVal = math.max(0, newVal)
    end

    if currentVal ~= newVal then
        recordTemporaryChange(triggerId, stateKey, newVal)
    end
    
    return newVal
end

-- 휴식, 훈련 등의 태그 처리
function processAuxiliaryTags(triggerId, text)
    if not text then return end
    
    local lang = getState(triggerId, "Language") or 0

    if text:match("<Rested>") then
        local oldAdr = getState(triggerId, "Adrenaline_Token_Count") or 0
        local oldAll = getState(triggerId, "All_or_Nothing_Token_Count") or 0
        local oldNoTime = getState(triggerId, "No_Time_to_Spare_Token_Count") or 0
        
        manageCombatState(triggerId, "Adrenaline", "set", 3)
        manageCombatState(triggerId, "AllOrNothing", "set", 3)
        manageCombatState(triggerId, "NoTime", "set", 2)
        
        -- 휴식 시 반전 토큰 및 실패 누적 초기화
        manageCombatState(triggerId, "Reversal", "set", 0)
        manageCombatState(triggerId, "FailureCount", "set", 0)
        
        if oldAdr < 3 or oldAll < 3 or oldNoTime < 2 then
            local msg = (lang == 1) 
                and "\n- System: 💤 충분한 휴식을 취했습니다. (토큰 회복)" 
                or "\n- System: 💤 You rested well. (Tokens Refilled)"
            addNotification(triggerId, msg)
        end
    end

    local trainExp = 0
    local trainReason = ""

    if text:match("<Hardly Trained>") then
        trainExp = 200
        trainReason = (lang == 1) and "고된 훈련" or "Hard Training"
    elseif text:match("<Trained>") then
        trainExp = 50
        trainReason = (lang == 1) and "훈련" or "Training"
    end

    if trainExp > 0 then
        -- Standalone RPG is always enabled
        local expBonusPercent = calculatePerkBonuses(triggerId, "exp_gain_modifier") or 0
        local finalAmount = trainExp
        if expBonusPercent > 0 then
            finalAmount = trainExp + math.floor(trainExp * (expBonusPercent / 100))
        end
        
        finalAmount = finalAmount or 0

        local currentEXP = getState(triggerId, "Character_EXP") or 0
        setState(triggerId, "Character_EXP", currentEXP + finalAmount)

        local safeReason = tostring(trainReason or "Training")
        local displayReason = safeReason .. " (+" .. tostring(finalAmount) .. ")"
        
        setChatVar(triggerId, "cv_exp_gained_this_turn", 1)
        setChatVar(triggerId, "cv_last_exp_amount", finalAmount)
        setChatVar(triggerId, "cv_last_exp_reason", displayReason)

        print("[Aux Tag] " .. safeReason .. " detected. Awarded " .. tostring(finalAmount) .. " EXP.")
    end
end

-- 스킬 체크 결과를 분석하여 Tempo 등의 퍽 효과를 적용
function processDiceResultToState(triggerId, chatHistory)
    local lastUserMsg = nil
    for i = #chatHistory - 1, 0, -1 do
        if chatHistory[i].role == "user" then
            lastUserMsg = chatHistory[i].data
            break
        end
    end
    if not lastUserMsg then return end

    local resultString = lastUserMsg:match("Result:%s*([^!%]]+)")
    if not resultString then return end
    
    local resultLower = resultString:lower()
    
    local isSuccess = resultLower:find("success") ~= nil
    local isFailure = resultLower:find("failure") ~= nil
    
    local skillName = lastUserMsg:match("<(%w+)%|") 

    -- 실패 누적 및 반전 토큰 로직
    if isFailure then
        manageCombatState(triggerId, "FailureCount", "add", 1)
        local currentFailures = getState(triggerId, "Failure_Accumulation_Count") or 0
        
        if currentFailures >= 2 then
            manageCombatState(triggerId, "FailureCount", "set", 0)
            manageCombatState(triggerId, "Reversal", "add", 1)
            
            local lang = getState(triggerId, "Language") or 0
            local msg = (lang == 1) 
                and "\n- System: 🔄 거듭된 실패 끝에 기회를 포착합니다. (반전 토큰 획득)" 
                or "\n- System: 🔄 After repeated failures, an opportunity arises. (Reversal Token Gained)"
            addNotification(triggerId, msg)
        end
    end

    local shouldMaintain = isSuccess

    if skillName == "Melee" and (getState(triggerId, "Perk_Tempo_Melee_acquired") or 0) == 1 then
        manageCombatState(triggerId, "Tempo_Melee", "set", shouldMaintain and 0 or 1)
        
    elseif skillName == "Marksmanship" and (getState(triggerId, "Perk_Tempo_Archery_acquired") or 0) == 1 then
        manageCombatState(triggerId, "Tempo_Archery", "set", shouldMaintain and 0 or 1)
    end

    if skillName == "Stealth" and isSuccess and (getState(triggerId, "Perk_Fearsome_Presence_acquired") or 0) == 1 then
        if (getState(triggerId, "State_Fearsome_Presence_Active") or 0) == 0 then
            manageCombatState(triggerId, "Fearsome", "set", 1)
        end
    end
    
    if lastUserMsg:find("%+2 Fearsome Presence") then
        manageCombatState(triggerId, "Fearsome", "set", 0)
    end
end

function resyncAllSkillBonuses(triggerId)
    print("--- Resyncing all skill bonuses based on the current bonus table ---")
    local somethingChanged = false
    for _, skillName in ipairs(skills) do
        local keyName = skillName:gsub(" ", "_")
        local currentPoints = getState(triggerId, keyName .. "_points") or 0
        local oldBonus = getState(triggerId, keyName .. "_bonus")

        local newBonus = 0
        for _, data in ipairs(skillPointBonusTable) do
            if currentPoints >= data.points then
                newBonus = data.bonus
                break
            end
        end

        if oldBonus ~= newBonus then
            setState(triggerId, keyName .. "_bonus", newBonus)
            somethingChanged = true
            print(string.format("  - '%s' bonus updated: %s -> %s", skillName, tostring(oldBonus), tostring(newBonus)))
        end
    end

    if not somethingChanged then
        print("--- All skill bonuses are already up-to-date. ---")
    end
end

end

function spendSkillPoint(triggerId, skillName)
    local availablePoints = getState(triggerId, "Skill_Points_Available") or 0
    local lang = getState(triggerId, "Language") or 0
    local charLevel = getState(triggerId, "Character_Level") or 1

    if availablePoints <= 0 then
        local message = (lang == 1) and "사용할 수 있는 스킬 포인트가 없습니다." or "No available skill points to spend."
        alertError(triggerId, message)
        return false
    end

    local keyName = skillName:gsub(" ", "_")
    local currentPoints = getState(triggerId, keyName .. "_points") or 0
    
    if currentPoints >= 7 then
        local message = (lang == 1) and (skillName .. " 스킬은 이미 최대치(Mythic)입니다.") or (skillName .. " skill is already at its maximum (Mythic).")
        alertError(triggerId, message)
        return false
    end

    if currentPoints == 5 and charLevel < 10 then
        local message = (lang == 1) and "이 등급(Legendary)에 도달하려면 캐릭터 레벨 10이 필요합니다." or "Character Level 10 is required to reach this rank (Legendary)."
        alertError(triggerId, message)
        return false
    end

    if currentPoints == 6 and charLevel < 14 then
        local message = (lang == 1) and "이 등급(Mythic)에 도달하려면 캐릭터 레벨 14가 필요합니다." or "Character Level 12 is required to reach this rank (Mythic)."
        alertError(triggerId, message)
        return false
    end

    local oldLevelName = getSkillLevelNameFromPoints(currentPoints)

    local newAvailablePoints = availablePoints - 1
    local newSkillPoints = currentPoints + 1
    
    setStateTracked(triggerId, "Skill_Points_Available", newAvailablePoints)
    setStateTracked(triggerId, keyName .. "_points", newSkillPoints)
    
    local newBonus = 0
    for _, data in ipairs(skillPointBonusTable) do
        if newSkillPoints >= data.points then
            newBonus = data.bonus
            break
        end
    end
    setStateTracked(triggerId, keyName .. "_bonus", newBonus)
    
    local newLevelName = getSkillLevelNameFromPoints(newSkillPoints)
    
    local successMessage = (lang == 1) and
        string.format("%s 스킬에 1 포인트를 투자했습니다. (현재 보너스: %+d)", skillName, newBonus) or
        string.format("Spent 1 point on %s skill. (Current Bonus: %+d)", skillName, newBonus)
    alertNormal(triggerId, successMessage)

    recordTemporaryChange(triggerId, "Skill_Points_Available", newAvailablePoints)
    recordTemporaryChange(triggerId, keyName .. "_points", newSkillPoints)
    recordTemporaryChange(triggerId, keyName .. "_bonus", newBonus)

    if oldLevelName ~= newLevelName then
        local chatIndex = getChatLength(triggerId) - 1
        if chatIndex >= 0 then
            local lastMessage = getChat(triggerId, chatIndex)
            local currentContent = lastMessage.data
            
            local infoString
            if lang == 1 then
                infoString = string.format("\n- Notification: 📜 여러가지 경험 덕분에 %s 기술이 발전했습니다: %s → %s.", skillName, oldLevelName, newLevelName)
            else
                infoString = string.format("\n- Notification: 📜 Thanks to various experiences, the %s skill has improved: %s → %s.", skillName, oldLevelName, newLevelName)
            end
            
            local newContent = currentContent .. infoString
            setChat(triggerId, chatIndex, newContent)
            
            return true
        end
    end
    
    return false
end

function checkForLevelUp(triggerId)
    local currentLevel = getState(triggerId, "Character_Level") or 1
    if currentLevel >= MAX_LEVEL then return end

    local currentEXP = getState(triggerId, "Character_EXP") or 0
    local levelData = levelUpTable[currentLevel]

    while currentLevel < MAX_LEVEL and currentEXP >= levelData.expRequired do
        currentLevel = currentLevel + 1
        currentEXP = currentEXP - levelData.expRequired
        
        local availablePoints = getState(triggerId, "Skill_Points_Available") or 0
        setStateTracked(triggerId, "Skill_Points_Available", availablePoints + levelData.pointsGained)

        local availablePerkPoints = getState(triggerId, "Perk_Points_Available") or 0
        setStateTracked(triggerId, "Perk_Points_Available", availablePerkPoints + (levelData.perkPointsGained or 0))
        
        local lang = getState(triggerId, "Language") or 0
        local message = (lang == 1) and
            string.format("\n- System: ⭐ 레벨업! 레벨 %d 달성! 스킬 포인트 %d, 특성 포인트 %d 획득!\n", currentLevel, levelData.pointsGained, (levelData.perkPointsGained or 0)) or
            string.format("\n- System: ⭐ LEVEL UP! Reached Lvl %d. Gained %d Skill Point(s) and %d Perk Point(s)!\n", currentLevel, levelData.pointsGained, (levelData.perkPointsGained or 0))
        
        addNotification(triggerId, message)
        
        if currentLevel < MAX_LEVEL then
            levelData = levelUpTable[currentLevel]
        end
    end
    
    setStateTracked(triggerId, "Character_Level", currentLevel)
    setStateTracked(triggerId, "Character_EXP", currentEXP)
end

function calculatePerkBonuses(triggerId, bonusType)
    local totalBonus = 0
    
    for perkId, perkData in pairs(perks) do
        if (getState(triggerId, "Perk_" .. perkId .. "_acquired") or 0) == 1 then
            if perkData.effects and perkData.effects[bonusType] then
                totalBonus = totalBonus + perkData.effects[bonusType]
            end
        end
    end

    return totalBonus
end

function acquirePerk(triggerId, perkId)
    local perkData = perks[perkId]
    local lang = getState(triggerId, "Language") or 0

    if not perkData then
        local message = (lang == 1) and "존재하지 않는 특성입니다." or "This perk does not exist."
        alertError(triggerId, message)
        return
    end

    local availablePoints = getState(triggerId, "Perk_Points_Available") or 0
    if availablePoints < perkData.cost then
        local message = (lang == 1) and ("특성 포인트가 부족합니다. (필요: " .. perkData.cost .. ")") or ("Not enough perk points. (Required: " .. perkData.cost .. ")")
        alertError(triggerId, message)
        return
    end

    local perkStateKey = "Perk_" .. perkId .. "_acquired"
    if (getState(triggerId, perkStateKey) or 0) == 1 then
        local message = (lang == 1) and "이미 습득한 특성입니다." or "Perk already acquired."
        alertError(triggerId, message)
        return
    end

    if perkData.requirements then
        local currentLevel = getState(triggerId, "Character_Level") or 1
        if perkData.requirements.level and currentLevel < perkData.requirements.level then
            local message = (lang == 1) and ("요구 레벨이 충족되지 않았습니다. (필요 레벨: " .. perkData.requirements.level .. ")") or ("Level requirement not met. (Required level: " .. perkData.requirements.level .. ")")
            alertError(triggerId, message)
            return
        end
        
        if perkData.requirements.skill_points then
            for skillName, requiredPoints in pairs(perkData.requirements.skill_points) do
                local keyName = skillName:gsub(" ", "_")
                local currentPoints = getState(triggerId, keyName .. "_points") or 0
                
                if currentPoints < requiredPoints then
                    local message = (lang == 1) and
                        string.format("%s 스킬 포인트가 부족합니다. (필요: %d, 현재: %d)", skillName, requiredPoints, currentPoints) or
                        string.format("Insufficient %s skill points. (Required: %d, Current: %d)", skillName, requiredPoints, currentPoints)
                    alertError(triggerId, message)
                    return
                end
            end
        end
    end
	
    if perkData.group then
        local groupName = perkData.group
        for otherPerkId, otherPerkData in pairs(perks) do
            if otherPerkId ~= perkId and otherPerkData.group == groupName then
                if (getState(triggerId, "Perk_" .. otherPerkId .. "_acquired") or 0) == 1 then
                    local perkDisplayName = perkId:gsub("_", " ")
                    local conflictingPerkDisplayName = otherPerkId:gsub("_", " ")
                    
                    local message = (lang == 1) and
                        string.format("'%s' 특성을 습득할 수 없습니다. 이미 같은 그룹의 '%s' 특성을 보유하고 있습니다.", perkDisplayName, conflictingPerkDisplayName) or
                        string.format("Cannot acquire '%s' perk. You already have the conflicting perk '%s' from the same group.", perkDisplayName, conflictingPerkDisplayName)
                    
                    alertError(triggerId, message)
                    return
                end
            end
        end
    end

    local newAvailablePoints = availablePoints - perkData.cost

    setState(triggerId, "Perk_Points_Available", newAvailablePoints)
    setChatVar(triggerId, "cv_Perk_Points_Available", newAvailablePoints)
    setState(triggerId, perkStateKey, 1)
    setChatVar(triggerId, "cv_" .. perkStateKey, 1)

    recordTemporaryChange(triggerId, "Perk_Points_Available", newAvailablePoints)
    recordTemporaryChange(triggerId, perkStateKey, 1)

    local perkDisplayNameForAlert = perkId:gsub("_", " ")
    local successMessage = (lang == 1) and ("특성 [" .. perkDisplayNameForAlert .. "] 을(를) 습득했습니다!") or ("Acquired the [" .. perkDisplayNameForAlert .. "] perk!")
    alertNormal(triggerId, successMessage)

    local protagonistName = getPersonaName(triggerId) or (lang == 1 and "당신" or "You")
    local perkDisplayName = perkId:gsub("_", " ")
    
    local infoString
    if lang == 1 then
        infoString = string.format("\n- Notification: ⭐ %s는 경험을 통해 **%s** 특성을 획득했습니다.", protagonistName, perkDisplayName)
    else
        infoString = string.format("\n- Notification: ⭐ %s has acquired the **%s** perk through experience.", protagonistName, perkDisplayName)
    end
    
    local chatIndex = getChatLength(triggerId) - 1
    if chatIndex >= 0 then
        local lastMessage = getChat(triggerId, chatIndex)
        local currentContent = lastMessage.data
        local newContent = currentContent .. infoString
        setChat(triggerId, chatIndex, newContent)
    end

    syncPerkVars(triggerId)
end

-- =======================================================
-- 자동 경험치 처리 시스템 (Auto EXP System)
-- =======================================================

function processSkillCheckEXP(triggerId)
    if (getState(triggerId, "RPG") or 0) ~= 1 then return end

    local chatLen = getChatLength(triggerId)
    local userMsgIndex = chatLen - 2
    if userMsgIndex < 0 then return end

    local userMsg = getChat(triggerId, userMsgIndex)
    if not userMsg or not userMsg.data then return end
    
    local content = userMsg.data

    -- 성공 문구가 있는지 확인
    if not content:find("Result:%s*Success") and 
       not content:find("Result:%s*Narrow Success") and 
       not content:find("Result:%s*Critical Success") then 
        return 
    end

    -- DC 숫자 추출 (vs 17 에서 17 추출)
    local baseDC = tonumber(content:match("vs%s+(%d+)"))
    if not baseDC or baseDC == 0 then return end

    -- 난이도 판정 및 경험치 양 결정
    local difficultyKey = getDifficultyFromDC(baseDC)
    local baseAmount = AUTO_EXP_VALUES[difficultyKey] or 100
    
    -- 보너스 계산
    local expBonusPercent = calculatePerkBonuses(triggerId, "exp_gain_modifier") or 0
    local finalAmount = math.floor(baseAmount * (1 + expBonusPercent / 100))

    -- 경험치 합산 및 저장
    local currentEXP = getState(triggerId, "Character_EXP") or 0
    setState(triggerId, "Character_EXP", currentEXP + finalAmount)

    -- 알림 변수 업데이트
    local skillName = content:match("<([^|]+)%|") or "Skill Check"
    local lang = getState(triggerId, "Language") or 0
    local reason = (lang == 1) and (skillName .. " 성공 (DC " .. baseDC .. ")") or (skillName .. " Success (DC " .. baseDC .. ")")
    
    setChatVar(triggerId, "cv_exp_gained_this_turn", 1)
    setChatVar(triggerId, "cv_last_exp_amount", finalAmount)
    setChatVar(triggerId, "cv_last_exp_reason", reason)
    
    print("[EXP 지급 완료] " .. reason .. " -> " .. finalAmount .. " EXP")
end

function callLLMWithRetry(triggerId, prompt, mode, callName)
    local MAX_RETRIES = 5
    local modelName = (mode == "1") and "Main" or "Auxiliary"
    local lastErrorReason = "Unknown error"

    for attempt = 1, MAX_RETRIES do
        print(string.format("[LLM Retry] Calling %s model for %s... (Attempt %d/%d)", modelName, callName, attempt, MAX_RETRIES))
        
        local response
        if mode == "1" then
            response = LLM(triggerId, prompt)
        else
            response = axLLM(triggerId, prompt)
        end
        
        if response and response.success then
            print(string.format("[LLM Retry] Success on attempt %d for %s.", attempt, callName))
            return response
        else
            if response and response.result then
                lastErrorReason = tostring(response.result)
            else
                lastErrorReason = "LLM did not return a valid response object."
            end
            print(string.format("[LLM Retry] FAILED on attempt %d for %s. Reason: %s", attempt, callName, lastErrorReason))
        end
    end

    print(string.format("[LLM Retry] All %d attempts failed for %s.", MAX_RETRIES, callName))
    
    local lang = getState(triggerId, "Language") or 0
    local userMessage
    if lang == 1 then
        userMessage = string.format(
            "LLM 호출 실패 (%s):\n\n최대 재시도 횟수(%d회)를 초과했습니다.\n마지막 오류:\n%s",
            callName, MAX_RETRIES, lastErrorReason
        )
    else
        userMessage = string.format(
            "LLM Call Failed (%s):\n\nExceeded max retries (%d attempts).\nLast Error:\n%s",
            callName, MAX_RETRIES, lastErrorReason
        )
    end
    alertError(triggerId, userMessage)

    return { success = false, result = "All " .. MAX_RETRIES .. " attempts failed for " .. callName .. ". Last Error: " .. lastErrorReason }
end

function BuildChatLog(triggerId, log)
    local MAX_MESSAGES = 6
    local logLength = #log
    local startIndex = math.max(1, logLength - MAX_MESSAGES + 1)
    
    local logsToAdd = {}
    local order = 1 
    
    for i = startIndex, logLength do
        local message = log[i]
        if message and type(message.data) == "string" and message.data ~= "" then
            local content = message.data

            content = content:gsub("%s*<[Cc]hoice>.-</[Cc]hoice>%s*", "")                        
                             :gsub("<!%-%-(.-)%-%->%s*\n?", "")

            if content:match("%S") then
                local role = (message.role == "user" and "USER" or "CHARACTER")
                table.insert(logsToAdd, string.format("<%d>%s: %s</%d>", order, role, content, order))
                order = order + 1
            end
        end
    end

    if #logsToAdd > 0 then
        local lastMsg = logsToAdd[#logsToAdd]
        local num, innerContent = lastMsg:match("^<(%d+)>(.*)</%1>$")
        if num and innerContent then
            logsToAdd[#logsToAdd] = string.format("<%s>Latest Scene\n%s</%s>", num, innerContent, num)
        end
    end

    return table.concat(logsToAdd, "\n\n")
end

local function generateOwnedPerksDescription(triggerId)
    local owned_perks_lines = {}

    for perkId, perkData in pairs(perks) do
        if (getState(triggerId, "Perk_" .. perkId .. "_acquired") or 0) == 1 then
            if perkData.prompt_desc and perkData.prompt_desc ~= "" then
                table.insert(owned_perks_lines, "- " .. perkData.prompt_desc)
            end
        end
    end

    if #owned_perks_lines > 0 then
        local header = "## Assistant Guidance: User's Perks\n- Perks currently possessed by {{user}}. When generating choices, utilize these perks appropriately for the situation.\n- Do not use the perk's name as the Skill Name.\n- Do not explicitly mention the name of the perk when generating choices. Only include how the effect of that perk is utilized.\n- Note which skill each perk belongs to."
        local perks_list_string = table.concat(owned_perks_lines, "\n")
        return header .. perks_list_string .. "\n"
    else
        return ""
    end
end

function executeChoiceReroll(triggerId)
    local chatLength = getChatLength(triggerId)
    local targetIndex = -1
    local originalContent = ""

    for i = chatLength - 2, 0, -1 do
        local msg = getChat(triggerId, i)
        if msg and msg.role == 'char' and msg.data:match("%[Skill Check%|") then
            targetIndex = i
            originalContent = msg.data
            break
        end
    end

    if targetIndex == -1 then
        error("스킬 체크가 포함된 AI 메시지를 찾을 수 없습니다.")
    end

    local contentWithoutChoices = originalContent
        :gsub("<[Cc]hoice>.-</[cC]hoice>", "")
        :gsub("<div class=\"choice%-container%-fallback\">.-</div>", "")
        :gsub("<div class=\"choice%-container\">.-</div>", "")
        :gsub("<div class=\"reroll%-button%-container\">.-</div>", "")
        :gsub("<%s*=[Ss][Aa][Vv][Ee]%|.-=%s*>", "")

    local chatHistoryForPrompt = getFullChat(triggerId)
    local logString = BuildChatLog(triggerId, { table.unpack(chatHistoryForPrompt, 1, chatLength - 1) })
    
    local choice_prompt = makeChoicePrompt(triggerId, logString)
    local choice_response = callLLMWithRetry(triggerId, choice_prompt, "2", "Choice Reroll")

    setState(triggerId, "user_choice_request", nil)

    if not choice_response or not choice_response.success then
        error("LLM 호출에 실패했습니다: " .. tostring(choice_response.result))
    end
    
    local new_choice_data = choice_response.result:gsub("```", ""):gsub("<[Tt]houghts>.-</[Tt]houghts>", "")
    
    local finalContent = contentWithoutChoices .. "\n" .. new_choice_data
    
    setChat(triggerId, targetIndex, finalContent)
    print("선택지 재생성을 완료하고 " .. targetIndex .. "번 메시지를 업데이트했습니다.")
end

rerollChoices = async(function(triggerId)
    local lang = getState(triggerId, "Language") or 0
    local rerollMessage

    if lang == 1 then
        rerollMessage = '<b>🔄 선택지를 재생성하는 중입니다... 메시지를 보내거나 리롤하지 마세요.</b>'
    else
        rerollMessage = '<b>🔄 Rerolling choices... Please do not send messages or reroll.</b>'
    end

    addChat(triggerId, 'system', rerollMessage)
    
    local success, err = pcall(executeChoiceReroll, triggerId)
    
    removeChat(triggerId, -1)
    
    if not success then
        local errorMessage = (lang == 1) and "선택지 재생성에 실패했습니다: " or "Failed to reroll choices: "
        alertError(triggerId, errorMessage .. tostring(err))
    end
end)

function executeForceSkillCheckLogic(triggerId)
    local lang = getState(triggerId, "Language") or 0
    local chatLength = getChatLength(triggerId)
    local targetIndex = -1
    local originalContent = ""

    for i = chatLength - 2, 0, -1 do
        local msg = getChat(triggerId, i)
        if msg and msg.role == 'char' then
            targetIndex = i
            originalContent = msg.data
            break
        end
    end

    if targetIndex == -1 then
        error((lang == 1) and "오류: 스킬 체크를 추가할 AI 응답을 찾을 수 없습니다." or "Error: Could not find an AI response to add the Skill Check to.")
    end

    -- 스킬 체크 생성을 위한 프롬프트 구성
    local chatHistoryForSkillCheck = getFullChat(triggerId)
    table.remove(chatHistoryForSkillCheck)
    local logString = BuildChatLog(triggerId, chatHistoryForSkillCheck)
    local skillCheckPrompt = makeSkillCheckPrompt(triggerId, logString)
    
    -- LLM 호출 (스킬 체크 생성)
    local skillCheckResponse = callLLMWithRetry(triggerId, skillCheckPrompt, "2", "Forced Skill Check")

    if not skillCheckResponse or not skillCheckResponse.success then
        error(((lang == 1) and "스킬 체크 생성에 실패했습니다: " or "Failed to generate Skill Check: ") .. tostring(skillCheckResponse.result))
    end
    
    local skillCheckString = skillCheckResponse.result:gsub("```", ""):gsub("<[Tt]houghts>.-</[Tt]houghts>", "")

    -- 생성된 스킬 체크에서 DC를 파싱
    local difficulty_keyword = skillCheckString:match("DC:%s*([%w%s]+)")
    if difficulty_keyword then

        difficulty_keyword = difficulty_keyword:gsub("%]", "")
        local key = difficulty_keyword:match("^%s*(.-)%s*$"):lower()
        
        local range = difficultyRanges[key]
        if range then
            local random_dc = math.random(range.min, range.max)
            setState(triggerId, "diceRoll_RequiredValue", random_dc)
            setChatVar(triggerId, "cv_diceRoll_RequiredValue", random_dc)
            print(string.format("[ForceSkillCheck] Difficulty keyword '%s' parsed. DC set to: %d (Range: %d-%d)", key, random_dc, range.min, range.max))
        else
            print("[ForceSkillCheck] Warning: Found DC keyword '" .. key .. "' but it has no defined range.")
        end
    end

    local contentWithSkillCheck = originalContent .. "\n\n" .. skillCheckString

    -- 선택지 생성에 관한 프롬프트 조합
    local chatHistoryForChoices = getFullChat(triggerId)
    table.remove(chatHistoryForChoices)
    chatHistoryForChoices[targetIndex + 1].data = contentWithSkillCheck
    
    local logString = BuildChatLog(triggerId, chatHistoryForChoices)
    local choicePrompt = makeChoicePrompt(triggerId, logString)
    
    -- LLM 호출
    local choiceResponse = callLLMWithRetry(triggerId, choicePrompt, "2", "Forced Choices")
    
    local finalContent = contentWithSkillCheck
    if choiceResponse and choiceResponse.success then
        local choiceString = choiceResponse.result:gsub("```", ""):gsub("<[Tt]houghts>.-</[Tt]houghts>", "")
        finalContent = finalContent .. "\n\n" .. choiceString
    else
        error(((lang == 1) and "선택지 생성에 실패했습니다: " or "Failed to generate choices: ") .. tostring(choiceResponse.result))
    end

    -- 최종 내용 업데이트
    setChat(triggerId, targetIndex, finalContent)
end

ForceSkillCheck = async(function(triggerId)
    local workingMessage = '<b>🔄 Generating Skill Check and Choices...</b>'
    addChat(triggerId, 'system', workingMessage)
    
    local success, err = pcall(executeForceSkillCheckLogic, triggerId)
    
    removeChat(triggerId, -1)
    
    if not success then
        alertError(triggerId, tostring(err))
    end
end)

function collectAllStateVariables(triggerId)
    local stateData = {}

    print("Collecting RPG System state...")
    stateData["Character_Level"] = getState(triggerId, "Character_Level") or 1
    stateData["Character_EXP"] = getState(triggerId, "Character_EXP") or 0
    stateData["Skill_Points_Available"] = getState(triggerId, "Skill_Points_Available") or 0
    stateData["Perk_Points_Available"] = getState(triggerId, "Perk_Points_Available") or 0

    stateData["State_Tempo_Melee_Lost"] = getState(triggerId, "State_Tempo_Melee_Lost") or 0
    stateData["State_Tempo_Archery_Lost"] = getState(triggerId, "State_Tempo_Archery_Lost") or 0
    
    stateData["State_Fearsome_Presence_Active"] = getState(triggerId, "State_Fearsome_Presence_Active") or 0
    stateData["Lucky_Coin_Count"] = getState(triggerId, "Lucky_Coin_Count") or 0
    stateData["Adrenaline_Token_Count"] = getState(triggerId, "Adrenaline_Token_Count") or 0 
    stateData["No_Time_to_Spare_Token_Count"] = getState(triggerId, "No_Time_to_Spare_Token_Count") or 0
    stateData["All_or_Nothing_Token_Count"] = getState(triggerId, "All_or_Nothing_Token_Count") or 0
    stateData["Preparation_Token_Count"] = getState(triggerId, "Preparation_Token_Count") or 0
    
    -- 반전 토큰 및 실패 누적 카운트 저장
    stateData["Reversal_Token_Count"] = getState(triggerId, "Reversal_Token_Count") or 0
    stateData["Failure_Accumulation_Count"] = getState(triggerId, "Failure_Accumulation_Count") or 0
    
    for perkId, _ in pairs(perks) do
        stateData["Perk_" .. perkId .. "_acquired"] = getState(triggerId, "Perk_" .. perkId .. "_acquired") or 0
    end

    for _, skillName in ipairs(skills) do
        local keyName = skillName:gsub(" ", "_")
        stateData[keyName .. "_points"] = getState(triggerId, keyName .. "_points") or 0
        stateData[keyName .. "_bonus"] = getState(triggerId, keyName .. "_bonus") or -1
    end

    return stateData
end

local stateChangeTracker = {}

function setStateTracked(triggerId, key, value)
    setState(triggerId, key, value)
    
    if not stateChangeTracker[triggerId] then
        stateChangeTracker[triggerId] = {}
    end
    stateChangeTracker[triggerId][key] = true
end

function batchSetStateTracked(triggerId, updates)
    if not stateChangeTracker[triggerId] then
        stateChangeTracker[triggerId] = {}
    end
    
    for key, value in pairs(updates) do
        setState(triggerId, key, value)
        stateChangeTracker[triggerId][key] = true
    end
end

function generateSaveString(triggerId)
    local stateTable = collectAllStateVariables(triggerId)
    
    local parts = {}
    local sortedKeys = {}
    for k in pairs(stateTable) do table.insert(sortedKeys, k) end
    table.sort(sortedKeys)
    
    for _, key in ipairs(sortedKeys) do
        local value = stateTable[key]
        table.insert(parts, string.reverse(key) .. ":" .. tostring(value))
    end

    return "<!--[SAVE|" .. table.concat(parts, "|") .. "]-->"
end

function parseAndApplySaveString(triggerId, saveString)
    local start_marker = saveString:find("<!--[SAVE|", 1, true)
    if not start_marker then return false end
    
    local end_marker = saveString:find("]-->", start_marker, true)
    if not end_marker then return false end
    
    local dataPart = saveString:sub(start_marker + 10, end_marker - 1)
    local len = #dataPart
    
    if len == 0 then return true end
    
    print("--- Applying State from Hidden SaveString ---")
    
    local stateUpdates = {}
    local updateCount = 0
    local pos = 1
    
    while pos <= len do
        local colonPos = dataPart:find(":", pos, true)
        if not colonPos then break end
        
        local pipePos = dataPart:find("|", colonPos, true)
        local endPos = pipePos or (len + 1)
        
        local reversedKey = dataPart:sub(pos, colonPos - 1)
        local key = string.reverse(reversedKey)
        
        local valueStr = dataPart:sub(colonPos + 1, endPos - 1)
        
        local finalValue
        local firstByte = valueStr:byte(1)
        
        if valueStr == "nil" then
            finalValue = nil
        elseif firstByte == 45 or (firstByte >= 48 and firstByte <= 57) then
            finalValue = tonumber(valueStr) or valueStr
        else
            finalValue = valueStr
        end
        
        stateUpdates[key] = finalValue
        updateCount = updateCount + 1
        pos = endPos + 1
    end
    
    batchSetStateTracked(triggerId, stateUpdates)
    
    print(string.format("--- State Apply Complete (%d updates) ---", updateCount))
    return true
end

function loadLatestStateFromHistory(triggerId)
    local stateWasLoaded = false
    local chatLength = getChatLength(triggerId)

    for i = chatLength - 1, 0, -1 do
        local message = getChat(triggerId, i)
        
        if message and message.role == "char" and message.data then
            local saveString = message.data:match("(<!%-%-%[SAVE%|.-%]%-%->)")
            if saveString then
                parseAndApplySaveString(triggerId, saveString)
                stateWasLoaded = true
                break 
            end
        end
    end
    
    local tempDelta = getState(triggerId, "temporary_state_delta")
    if tempDelta and tempDelta.data then
        for key, value in pairs(tempDelta.data) do
            setState(triggerId, key, value)
        end
    end
end

function syncCharacterAndSkillVars(triggerId)

    smartSetChatVar(triggerId, "cv_Character_Level", level)

-- 퍽 동기화

    smartSetChatVar(triggerId, "cv_Fearsome_Presence_Active", getState(triggerId, "State_Fearsome_Presence_Active") or 0)

    setState(triggerId, "has_any_perk", isAnyPerkAcquired)
    smartSetChatVar(triggerId, "cv_has_any_perk", isAnyPerkAcquired)

    if not text then return end

    if text:match("Sacrificed one Lucky Token") or text:match("행운의 토큰을 하나 희생하였다") then
        manageCombatState(triggerId, "LuckyCoin", "use", 1)

        local amount = tonumber(text:match("%+(%d+) Preparation")) or 1
        manageCombatState(triggerId, "Preparation", "use", amount)
    end

    if text:match("forces a success but sustains") or text:match("억지로 성공시키는 대신") then
        manageCombatState(triggerId, "NoTime", "use", 1)
    end

        manageCombatState(triggerId, "AllOrNothing", "use", 1)
    end

    if text:match("Used Reversal Token") or text:match("상황이 반전된다.") then
        manageCombatState(triggerId, "Reversal", "use", 1)
    end
end

function initializeSession(triggerId)
    local is_session_initialized = getState(triggerId, "session_initialized")
    local load_success, load_err = pcall(loadLatestStateFromHistory, triggerId)
    if not load_success then
        print("[ERROR] loadLatestStateFromHistory failed: " .. tostring(load_err))
    end
   
    local resync_success, resync_err = pcall(resyncAllSkillBonuses, triggerId)
    if not resync_success then
        print("[ERROR] resyncAllSkillBonuses failed: " .. tostring(resync_err))
    end

    if not is_session_initialized then
        local cv_RPG = getChatVar(triggerId, "cv_RPG")
        if cv_RPG ~= "0" then
            setState(triggerId, "RPG", 1)
            setChatVar(triggerId, "cv_RPG", 1)
        else
            setState(triggerId, "RPG", 0)
        end
        
        initializeCharacterAndSkillSystem(triggerId)
        
        setState(triggerId, "session_initialized", true)
    end
    
    if getState(triggerId, "Language") == nil then
        setState(triggerId, "Language", 0) 
    end
    
    setChatVar(triggerId, "cv_Language", getState(triggerId, "Language"))
end

        else isSuccess = false; outcome = (targetValue - finalDiceResult) <= 2 and "Narrow Failure" or "Failure" end

        if isSuccess and (getState(id, "Perk_Preparation_acquired") or 0) == 1 then
            manageCombatState(id, "Preparation", "add", 1)
        end
        
        local modifier_display = string.format("%+d", modifier)
		
        local line1 = string.format("<%s|%s|%s|%s|Difficulty Modifier:%s>", actionName, description, tooltip1, tooltip2, modifier_display)

            local currentTurn = getChatLength(id)
            local function canTriggerPerk(perkName)
                 local lastTurn = getState(id, "Cooldown_" .. perkName) or -100
                 if (currentTurn - lastTurn) > 2 then
                     setState(id, "Cooldown_" .. perkName, currentTurn)
                     return true
                 end
                 return false
            end
            
            if actionName == "Melee" then
                if (getState(id, "Perk_Footwork_acquired") or 0) == 1 and canTriggerPerk("Footwork") then perkText=perkText.."\n<!--Footwork activated. In the next response, instead of describing {{user}} getting hit, generate a Skill Check for evading the attack.-->" end
                if (getState(id, "Perk_Bulwark_acquired") or 0) == 1 and canTriggerPerk("Bulwark") then perkText=perkText.."\n<!--Bulwark activated. In the next response, instead of describing {{user}} getting hit, generate a Skill Check for blocking the attack.-->" end
            elseif actionName == "Marksmanship" then

                if (getState(id, "Perk_Tactical_Redeployment_acquired") or 0) == 1 and canTriggerPerk("Tactical_Redeployment") then perkText=perkText.."\n<!--Tactical Redeployment activated.             elseif actionName == "Stealth" then

In the next response, instead of describing {{user}} getting caught or failing stealth, generate a Skill Check for escaping the situation and re-entering stealth.-->" end

function main_rpg(triggerId)
    initializeSession(triggerId)

    local chatHistory = getFullChat(triggerId)
    if #chatHistory == 0 then return end
    local latestMessage = chatHistory[#chatHistory]
    local mainModelResponse = latestMessage.data

    if mainModelResponse then
        local difficulty_keyword = mainModelResponse:match("DC:%s*([%w%s]+)")
        if difficulty_keyword then
             local key = difficulty_keyword:match("^%s*(.-)%s*$"):lower()
             local range = difficultyRanges[key]
             if range then
                 local random_dc = math.random(range.min, range.max)
                 setState(triggerId, "diceRoll_RequiredValue", random_dc)
                 setChatVar(triggerId, "cv_diceRoll_RequiredValue", random_dc)
             end
        end
    end
    
    local cleanedRP = mainModelResponse
        :gsub("%s*<[Tt]houghts>.-</[Tt]houghts>%s*\n?", "")
        :gsub("%s*<[Tt]hink>.-</[Tt]hink>%s*\n?", "")
        :gsub("%s*$", "")
 
    if cleanedRP:match("<[Qq]uest board>") then
        cleanedRP = cleanedRP:gsub("%s*%[Skill Check%|.-%|DC:.-%]%s*", "")
    end

    local choices_needed = cleanedRP:match("%[Skill Check%|.-%|DC:.-%]")
    
    setChatVar(triggerId, "cv_last_exp_reason", "")
    setChatVar(triggerId, "cv_exp_gained_this_turn", 0)	
    setChatVar(triggerId, "cv_last_exp_amount", 0)	

    local lastUserMsg = nil
    for i = #chatHistory, 1, -1 do
        if chatHistory[i].role == "user" then
            lastUserMsg = chatHistory[i].data
            break
        end
    end

	processTokenUsage(triggerId, lastUserMsg)
    processSkillCheckEXP(triggerId) 
    processAuxiliaryTags(triggerId, mainModelResponse)
	 
    local choice_data = ""
    if choices_needed then
        local tempChatHistory = getFullChat(triggerId)
        tempChatHistory[#chatHistory].data = cleanedRP
        local LogStringForPrompts = BuildChatLog(triggerId, tempChatHistory)
        
        local choice_prompt = makeChoicePrompt(triggerId, LogStringForPrompts)
        -- mode "2" is usually Auxiliary. Using "2" as per original context.
        local choice_response = callLLMWithRetry(triggerId, choice_prompt, "2", "Choice Generation")
        if choice_response.success and type(choice_response.result) == "string" then
            choice_data = choice_response.result:gsub("```", ""):gsub("<[Tt]houghts>.-</[Tt]houghts>", "")
        else
            local lang = getState(triggerId, "Language") or 0
            local fallbackMessage = (lang == 1) and "선택지 생성에 실패했습니다." or "Failed to generate choices."
            choice_data = '<div class="choice-container-fallback">'..fallbackMessage..'</div>'
            alertError(triggerId, "[RPG System] Failed to generate choices: " .. tostring(choice_response.result))
        end
    end
    
    checkForLevelUp(triggerId)
    
    local notificationBlock = ""
    if session_notifications_manager[triggerId] and #session_notifications_manager[triggerId] > 0 then
        notificationBlock = "\n" .. table.concat(session_notifications_manager[triggerId], "\n")
        session_notifications_manager[triggerId] = {}
    end

    processDiceResultToState(triggerId, chatHistory)

    local statusBlock = (choice_data ~= "" and "\n" .. choice_data or "")

    local finalVisibleMessage = cleanedRP .. notificationBlock .. "\n\n\n" .. statusBlock

    local tempDelta = getState(triggerId, "temporary_state_delta")
    if tempDelta and tempDelta.data then
        for k, v in pairs(tempDelta.data) do
            setState(triggerId, k, v)
        end
    end

    local saveString = generateSaveString(triggerId)
    local finalMessageWithSaveData = finalVisibleMessage .. "\n" .. saveString
    setChat(triggerId, #chatHistory - 1, finalMessageWithSaveData)

    if getState(triggerId, "temporary_state_delta") then
        setState(triggerId, "temporary_state_delta", nil)
    end
    
    syncCharacterAndSkillVars(triggerId)
	syncPerkVars(triggerId)
end

onOutput = async(function(triggerId)
    local success, result = pcall(main_rpg, triggerId)
    if not success then
        alertError(triggerId, "[RPG System] Backend Error: " .. tostring(result))
    end
end)

-- [공통] 화면 강제 새로고침 함수
function ForceRefresh(triggerId)

    end

Cheat_ResetDC = async(function(triggerId)
    local lang = getState(triggerId, "Language") or 0
    
    local options = {
        "🎲 Easy (쉬움)",
        "🎲 Normal (보통)",
        "🎲 Hard (어려움)",
        "🎲 Very Hard (매우 어려움)",
        "---",
        "🔃 Regenerate Choices (선택지 재생성)",
        "✨ Request Specific Choice (선택지 요청)",
        "🗑️ Delete Skill Check (스킬 체크 삭제)"
    }
    
    local indexMap = {
        [1] = "easy",
        [2] = "normal",
        [3] = "hard",
        [4] = "very hard"
    }

    local selectedIndex_str = alertSelect(triggerId, options):await()

    if not selectedIndex_str then
        return
    end
    
    local selectedIndex = tonumber(selectedIndex_str) + 1

    if selectedIndex == 5 then
        return
    elseif selectedIndex == 6 then
        local alertMsg = (lang == 1) and "🪶 선택지를 재생성 중입니다. 새 메시지를 보내거나 리롤하지 마세요." or "🪶 Regenerating choices. Please do not send messages or reroll."
        alertNormal(triggerId, alertMsg)
        rerollChoices(triggerId):await()
        return
    elseif selectedIndex == 7 then
        local prompt = (lang == 1) and "🪶 원하는 선택지의 내용이나 방향을 입력하세요." or "🪶 Enter the desired content or direction for the choices."
        local requestInput = alertInput(triggerId, prompt):await()
        
        if requestInput and requestInput ~= "" then
            setState(triggerId, "user_choice_request", requestInput)
            local alertMsg = (lang == 1) and "🪶 요청을 반영하여 선택지를 재생성 중입니다..." or "🪶 Regenerating choices with your request..."
            alertNormal(triggerId, alertMsg)
            rerollChoices(triggerId):await()
        end
        return
    elseif selectedIndex == 8 then
        local chatLength = getChatLength(triggerId)
        local targetIndex = -1

        for i = chatLength - 1, 0, -1 do
            local msg = getChat(triggerId, i)
            if msg and msg.role == 'char' and msg.data:match("%[Skill Check%|") then
                targetIndex = i
                break
            end
        end

        if targetIndex ~= -1 then
            local originalContent = getChat(triggerId, targetIndex).data
            
            local cleanedContent = originalContent
                :gsub("%[Skill Check%|.-%]%s*\n?", "")
                :gsub("<[Cc]hoice>.-</[Cc]hoice>%s*\n?", "")
                :gsub('<div class="choice%-container">.-</div>%s*\n?', "")
                :gsub('<div class="choice%-container%-fallback">.-</div>%s*\n?', "")
                :gsub('<div class="reroll%-button%-container">.-</div>%s*\n?', "")

            setChat(triggerId, targetIndex, cleanedContent)
            
            local successMessage = (lang == 1) and "가장 최근의 스킬 체크와 선택지를 제거했습니다." or "Successfully removed the most recent skill check and choices."
            alertNormal(triggerId, successMessage)
        else
            local errorMessage = (lang == 1) and "제거할 스킬 체크를 찾을 수 없습니다." or "Could not find a skill check to remove."
            alertError(triggerId, errorMessage)
        end
        return
    end
    
    local key = indexMap[selectedIndex]
    
    if key then
        local range = difficultyRanges[key]
        
        if range then
            local newDC = math.random(range.min, range.max)

            setState(triggerId, "diceRoll_RequiredValue", newDC)
            setChatVar(triggerId, "cv_diceRoll_RequiredValue", newDC)

            local message = (lang == 1) and
                ("DC 값이 " .. newDC .. "(으)로 재설정되었습니다. (난이도: " .. key .. ")") or
                ("DC value has been reset to " .. newDC .. ". (Difficulty: " .. key .. ")")
            alertNormal(triggerId, message)
        else
            alertError(triggerId, "Error: Could not find difficulty range for key '" .. key .. "'.")
        end
    end
end)

function resetCharacterSkillsAndPerks(triggerId)
    for _, skillName in ipairs(skills) do
        local keyName = skillName:gsub(" ", "_")
        setState(triggerId, keyName .. "_points", 0)
        recordTemporaryChange(triggerId, keyName .. "_points", 0)
        
        local initialBonus = 0
        for _, data in ipairs(skillPointBonusTable) do
            if 0 >= data.points then
                initialBonus = data.bonus
                break
            end
        end
        setState(triggerId, keyName .. "_bonus", initialBonus)
        recordTemporaryChange(triggerId, keyName .. "_bonus", initialBonus)
    end
    
    for perkId, _ in pairs(perks) do
        setState(triggerId, "Perk_" .. perkId .. "_acquired", 0)
        recordTemporaryChange(triggerId, "Perk_" .. perkId .. "_acquired", 0)
    end
end

SetInitialCharacterLevel = async(function(triggerId)
    local promptMessage = string.format("설정할 초기 레벨을 입력하세요 (1-%d):", MAX_LEVEL)
    
    local userInput = alertInput(triggerId, promptMessage):await()

    if not userInput or userInput == "" then
        return
    end

    local targetLevel = tonumber(userInput)

    if not targetLevel or targetLevel < 1 or targetLevel > MAX_LEVEL then
        local lang = getState(triggerId, "Language") or 0
        local errorMessage = (lang == 1) and
            string.format("잘못된 입력입니다. 1에서 %d 사이의 숫자만 입력해주세요.", MAX_LEVEL) or
            string.format("Invalid input. Please enter a number between 1 and %d.", MAX_LEVEL)
        alertError(triggerId, errorMessage)
        return
    end
    
    targetLevel = math.floor(targetLevel)

    resetCharacterSkillsAndPerks(triggerId)

    local totalSkillPoints = 0
    local totalPerkPoints = 0

    if targetLevel > 1 then
        for i = 1, targetLevel - 1 do
            local levelData = levelUpTable[i]
            if levelData then
                totalSkillPoints = totalSkillPoints + (levelData.pointsGained or 0)
                totalPerkPoints = totalPerkPoints + (levelData.perkPointsGained or 0)
            end
        end
    end
    
    setState(triggerId, "Character_Level", targetLevel)
    setState(triggerId, "Character_EXP", 0)
    setState(triggerId, "Skill_Points_Available", totalSkillPoints)
    setState(triggerId, "Perk_Points_Available", totalPerkPoints)
    
    recordTemporaryChange(triggerId, "Character_Level", targetLevel)
    recordTemporaryChange(triggerId, "Character_EXP", 0)
    recordTemporaryChange(triggerId, "Skill_Points_Available", totalSkillPoints)
    recordTemporaryChange(triggerId, "Perk_Points_Available", totalPerkPoints)
    
    syncAllChatVars(triggerId)
    
    local lang = getState(triggerId, "Language") or 0
    local message = (lang == 1) and
        string.format("캐릭터를 레벨 %d(으)로 설정했습니다.\n스킬 포인트: %d\n특성 포인트: %d", targetLevel, totalSkillPoints, totalPerkPoints) or
        string.format("Character set to Level %d.\nSkill Points: %d\nPerk Points: %d", targetLevel, totalSkillPoints, totalPerkPoints)
    alertNormal(triggerId, message)

end)

function TogglePanel_Skill(triggerId)
    local currentState = getChatVar(triggerId, "ui_status_skill")
    if currentState == "opened" then
        setChatVar(triggerId, "ui_status_skill", "")
    else
        setChatVar(triggerId, "ui_status_skill", "opened")
        setChatVar(triggerId, "ui_status_setting", "")			
    end

end

    local currentState = getChatVar(triggerId, "ui_status_setting")
    if currentState == "opened" then
        setChatVar(triggerId, "ui_status_setting", "")		
    else
        setChatVar(triggerId, "ui_status_setting", "opened")
        setChatVar(triggerId, "ui_status_skill", "")		
    end
    ForceRefresh(triggerId)
end

-- [공통] 화면 강제 새로고침 함수
function ForceRefresh(triggerId)
    local lastIndex = getChatLength(triggerId) - 1
    if lastIndex >= 0 then
        local lastMsg = getChat(triggerId, lastIndex)
        setChat(triggerId, lastIndex, lastMsg.data) 
    end
end
